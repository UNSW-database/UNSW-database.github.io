package org.dkr.graph.algo.utils;
import org.apache.commons.cli.*;
import org.apache.commons.io.FileUtils;

import java.io.File;
import java.io.FileNotFoundException;
import java.io.IOException;
import java.nio.charset.StandardCharsets;
import java.util.HashMap;
import java.util.Scanner;

/**
 * this class is used for raw graph process
 */
public class GraphProcessor {

    /**
     * this function is used to process the raw graph
     * @param inputFile input file path
     * @param outFile output file path
     * @param skip skipped how many lines at the front of input file
     */
    static void Process(String inputFile, String outFile, String split, int skip) {

        int numNodes = 0;
        StringBuilder edgeResult = new StringBuilder();
        StringBuilder vertexResult = new StringBuilder();
        HashMap<String, String> map = new HashMap<>();

        vertexResult.append(String.format("%s %s\n", numNodes, numNodes));

        try {
            File myObj = new File(inputFile);
            Scanner myReader = new Scanner(myObj);

            while (myReader.hasNextLine()) {
                // skip the useless information
                if (skip > 0) {
                    skip --;
                    myReader.nextLine();
                    continue;
                }

                String edge = myReader.nextLine();
                String[] nodes = edge.split(split);

                for (String node : nodes) {
                    if (!map.containsKey(node)) {
                        map.put(node, String.valueOf(node));
                        numNodes += 1;

                        vertexResult.append(String.format("%s %s\n", numNodes, numNodes));
                    }
                }

                edgeResult.append(String.format("%s %s\n", map.get(nodes[0]), map.get(nodes[1])));
            }
            myReader.close();

            // write edges file
            FileUtils.writeStringToFile(new File(outFile + ".e"), edgeResult.toString(),
                    StandardCharsets.UTF_8);

            // write vertex file
            FileUtils.writeStringToFile(new File(outFile + ".v"), vertexResult.toString(),
                    StandardCharsets.UTF_8);

        } catch (FileNotFoundException e) {
            System.out.println("An error occurred.");
            e.printStackTrace();
        } catch (IOException e) {
            System.out.println("Write file error.");
        }
    }

    public static void main(String[] args) {
        Options options = new Options();

        Option input = new Option("i", "Input", true, "Input raw graph");
        input.setRequired(true);

        Option output = new Option("o", "Output", true, "Output graph filename");
        output.setRequired(false);

        Option skip = new Option("s", "Skipped", true, "Skip how many lines");
        skip.setRequired(false);

        Option split = new Option("sp", "Split", true, "Split in the raw graph");
        skip.setRequired(true);

        options.addOption(input);
        options.addOption(output);
        options.addOption(skip);
        options.addOption(split);

        CommandLineParser parser = new DefaultParser();
        HelpFormatter formatter = new HelpFormatter();
        CommandLine cmd = null;

        try {
            cmd = parser.parse(options, args);
        } catch (ParseException e) {
            System.out.println(e.getMessage());
            formatter.printHelp("utility-name", options);
            System.exit(1);
        }

        String inputFilePath = cmd.getOptionValue("Input");
        String outputFilePath = cmd.getOptionValue("Output");
        String skipLines = cmd.getOptionValue("Skipped");
        String spliter = cmd.getOptionValue("Split");

        // check whether is null, if null then set the default value
        if (outputFilePath == null) outputFilePath = inputFilePath;
        if (skipLines == null) skipLines = "0";



        Process(inputFilePath, outputFilePath, spliter, Integer.parseInt(skipLines));

    }
}
