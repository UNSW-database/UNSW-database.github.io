package org.dkr.graph.algo.utils;

/**
 * Type parser to parse the values
 * @param <T> value type
 */
public class TypeParser <T> {

    /**
     * give a value, parse it as the specific type
     * @param type value type
     * @param value value
     * @return parsed value
     */
    public T parse(Class<T> type, String value) {
        T val = null;

        if (type.equals(Integer.class)) {
            val = (T) Integer.valueOf(value);
        } else if (type.equals(Long.class)) {
            val = (T) Long.valueOf(value);
        } else if (type.equals(String.class)) {
            val = (T) String.valueOf(value);
        } else if (type.equals(Float.class)) {
            val = (T) Float.valueOf(value);
        }  else if (type.equals(Double.class)) {
            val = (T) Double.valueOf(value);
        }  else if (type.equals(Short.class)) {
            val = (T) Short.valueOf(value);
        } else if (type.equals(Boolean.class)) {
            val = (T) Boolean.valueOf(value);
        }

        return val;
    }
}
