package org.dkr.graph.io.utils;

import org.apache.flink.api.common.functions.FlatMapFunction;
import org.apache.flink.api.java.tuple.Tuple2;
import org.apache.flink.graph.Vertex;
import org.apache.flink.util.Collector;
import org.dkr.graph.algo.utils.TypeParser;

/**
 * Vertex splitter implement FlatMapFunction
 * @param <K> vertex id data type
 * @param <VV> vertex value data type
 */
public class VertexSplitter<K, VV> implements FlatMapFunction<String, Tuple2<K, VV>> {
    private final Class<K> vertexType;
    private final Class<VV> valueType;

    public VertexSplitter(Class<K> vertexType, Class<VV> valueType) {
        this.vertexType = vertexType;
        this.valueType = valueType;
    }

    @Override
    public void flatMap(String line, Collector<Tuple2<K, VV>> out) {
        String[] edge = line.split(" ");
        out.collect(new Tuple2<K, VV>(new TypeParser<K>().parse(vertexType, edge[0]),
                new TypeParser<VV>().parse(valueType, edge[1])));
    }
}
