package org.dkr.graph.io.utils;

import org.apache.flink.api.common.functions.FlatMapFunction;
import org.apache.flink.api.java.tuple.Tuple3;
import org.apache.flink.graph.Edge;
import org.apache.flink.util.Collector;
import org.dkr.graph.algo.utils.GraphType;
import org.dkr.graph.algo.utils.TypeParser;

/**
 * Edge splitter implement FlatMapFunction
 * @param <K> vertex id data type
 * @param <EV> edge value data type
 */
public class EdgeSplitter<K, EV> implements FlatMapFunction<String, Tuple3<K, K,  EV>> {
    private final Class<K> vertexType;
    private final Class<EV> edgeValueType;
    private final GraphType type;

    /**
     * constructor
     * @param vertexType vertex id data type class, e.g., Long.class
     * @param edgeValueType edge value data type class
     * @param type graph type
     */
    public EdgeSplitter(Class<K> vertexType, Class<EV> edgeValueType, GraphType type) {
        this.vertexType = vertexType;
        this.edgeValueType = edgeValueType;
        this.type = type;
    }

    @Override
    public void flatMap(String line, Collector<Tuple3<K, K,  EV>> out) {

        String[] edge = line.split(" ");

        if (this.type == GraphType.WEIGHTED) {
            out.collect(new Tuple3<K, K,  EV>(new TypeParser<K>().parse(vertexType, edge[0]),
                    new TypeParser<K>().parse(vertexType, edge[1]),
                    new TypeParser<EV>().parse(edgeValueType, edge[2])));
            out.collect(new Tuple3<K, K,  EV>(new TypeParser<K>().parse(vertexType, edge[1]),
                    new TypeParser<K>().parse(vertexType, edge[0]),
                    new TypeParser<EV>().parse(edgeValueType, edge[2])));
        } else if (this.type == GraphType.DIRECT) {
            out.collect(new Tuple3<K, K,  EV>(new TypeParser<K>().parse(vertexType, edge[0]),
                    new TypeParser<K>().parse(vertexType, edge[1]),
                    new TypeParser<EV>().parse(edgeValueType, "0")));
        } else if (this.type == GraphType.UNDIRECT) {
            out.collect(new Tuple3<K, K,  EV>(new TypeParser<K>().parse(vertexType, edge[0]),
                    new TypeParser<K>().parse(vertexType, edge[1]),
                    new TypeParser<EV>().parse(edgeValueType, "0")));
            out.collect(new Tuple3<K, K,  EV>(new TypeParser<K>().parse(vertexType, edge[1]),
                    new TypeParser<K>().parse(vertexType, edge[0]),
                    new TypeParser<EV>().parse(edgeValueType, "0")));
        }
    }
}
