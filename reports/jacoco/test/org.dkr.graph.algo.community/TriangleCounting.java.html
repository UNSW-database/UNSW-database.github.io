package org.dkr.graph.algo.community;

import org.apache.flink.graph.Graph;
import org.apache.flink.graph.GraphAlgorithm;
import org.apache.flink.graph.library.TriangleEnumerator;

/**
 * The class TriangleCounting enumerates unique triangles present in the input graph.
 *
 * @param <K> ID type
 * @param <VV> vertex value type
 * @param <EV> edge value type
 *
 * TODO: for triangle counting, we may do not need to enumerate all triangles.
 */
public class TriangleCounting<K extends Comparable<K>, VV, EV>
        implements GraphAlgorithm<K, VV, EV, Long> {

    public Long run(Graph<K, VV, EV> graph) throws Exception {
        return graph.run(new TriangleEnumerator<>()).count();
    }

}
