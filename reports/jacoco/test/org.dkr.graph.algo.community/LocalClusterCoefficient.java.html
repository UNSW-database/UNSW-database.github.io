package org.dkr.graph.algo.community;

import org.apache.flink.api.java.DataSet;
import org.apache.flink.graph.Graph;
import org.apache.flink.graph.GraphAlgorithm;
import org.apache.flink.graph.library.clustering.directed.LocalClusteringCoefficient;
import org.apache.flink.types.CopyableValue;

/**
 * The class LocalClusterCoefficient measures the connectedness of each vertex's neighborhood.
 *
 * @param <K> ID type
 * @param <VV> vertex value type
 * @param <EV> edge value type
 */
public class LocalClusterCoefficient<K extends Comparable<K> & CopyableValue<K>, VV, EV>
        implements GraphAlgorithm<K, VV, EV, DataSet<LocalClusteringCoefficient.Result<K>>> {

    public DataSet<LocalClusteringCoefficient.Result<K>> run(Graph<K, VV, EV> graph) throws Exception {
        LocalClusteringCoefficient<K,VV,EV> localClusteringCoefficient=new LocalClusteringCoefficient<>();
        return localClusteringCoefficient.runInternal(graph);
    }
}
