package org.dkr.graph.algo.community;

import org.apache.flink.api.java.tuple.Tuple2;
import org.apache.flink.api.java.tuple.Tuple6;

import java.util.HashSet;

/**
 * Data Type for StronglyConnectedComponents.
 *
 * <p>Tuple6<Long, Long, Long, Long, HashSet<Long>, HashSet<Long>> color, sccTag, minForward,
 * minBackward, inNbr, outNbr
 *
 * <p>For color: (i)color=(-1, -1), means not assigned; (ii)(color, color), color>=0, means founded;
 * (iii)otherwise, (x, y), means in processing.
 *
 * <p>For sccTag: (i)sccTag = false, means not assigned; (ii)sccTag = true, means founded.
 *
 * <p>For minForward & minBackward: -1, means Uninit.
 */
public class SCCVertexDataType
    extends Tuple6<Tuple2<Long, Long>, Boolean, Long, Long, HashSet<Long>, HashSet<Long>> {

  public SCCVertexDataType() {
    this(new Tuple2<>(-1L, -1L), false, -1L, -1L, new HashSet<Long>(), new HashSet<Long>());
  }

  public SCCVertexDataType(
      Tuple2<Long, Long> value0,
      Boolean value1,
      Long value2,
      Long value3,
      HashSet<Long> value4,
      HashSet<Long> value5) {
    this.f0 = value0;
    this.f1 = value1;
    this.f2 = value2;
    this.f3 = value3;
    this.f4 = value4;
    this.f5 = value5;
  }
}
