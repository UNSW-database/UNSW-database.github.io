package org.dkr.graph.algo.similarity;

import java.util.List;

/**
 * The CosineSimilaritySingleSource computes the cosine similarity (the cosine similarity between
 * two sets A and B is computed by similarity(A,B)=(Aè·¯B)/(|A|*|B|)) between vertices and the given
 * source vertex, then reports the top K results {@link DataSet}. It is implemented by reference
 * JaccardSimilaritySingleSource {@link JaccardSimilaritySingleSource} algorithm. The input graph
 * must be a simple graph containing no duplicate edges or self-loops.
 *
 * @param <K> graph id type
 * @param <VV> vertex value type
 * @param <EV> edge value type
 */
public class CosineSimilaritySingleSource<K extends Comparable<? super K>, VV, EV>
    extends JaccardSimilaritySingleSource<K, VV, EV> {
  public CosineSimilaritySingleSource(K srcVertexId, Integer topK) {
    super(srcVertexId, topK);
    setMeasure(new CosineSimilarityMeasure<>());
  }

  /**
   * Measure the cosine similarity between two lists
   *
   * @param <K> graph id type
   */
  public static class CosineSimilarityMeasure<K extends Comparable<? super K>>
      extends SimilarityMeasure<K> {
    public Double similarityCompute(List<K> srcNbr, List<K> dstNbr) {
      return commonNeighbor(srcNbr, dstNbr) / (Math.sqrt(srcNbr.size()) * Math.sqrt(dstNbr.size()));
    }
  }
}
