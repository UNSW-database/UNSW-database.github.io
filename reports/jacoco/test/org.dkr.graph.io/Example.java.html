package org.dkr.graph.io;

import org.apache.flink.api.java.ExecutionEnvironment;
import org.apache.flink.graph.Graph;
import org.dkr.graph.algo.utils.GraphType;

public class Example {

  public static void main(String[] args) throws Exception {
    ExecutionEnvironment env = ExecutionEnvironment.getExecutionEnvironment();

    GraphLoader<Long, Long, Long> graphLoader = new GraphLoader<>("data\\taro.v", "data\\taro.e",
            env, Long.class, Long.class, Long.class);

    Graph<Long, Long, Long> graph = graphLoader.load(GraphType.UNDIRECT);

    graph.getEdges().print();
  }
}
