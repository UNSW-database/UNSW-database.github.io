package org.dkr.graph.io;

import org.apache.flink.api.java.ExecutionEnvironment;

/**
 * The graph base for different kind graph types
 * @param <K> the vertex id data type
 * @param <VV> the vertex value data type
 * @param <EV> the edge value data type
 */
public class GraphBase<K, VV, EV> {
    public final String vertexPath, edgePath;
    public final ExecutionEnvironment env;
    public final Class<K> vertexType;
    public final Class<VV> valueType;
    public final Class<EV> edgeValueType;

    public GraphBase(String vertexPath, String edgePath, ExecutionEnvironment env,
                       Class<K> vertexType, Class<VV> valueType, Class<EV> edgeValueType) {
        this.vertexPath = vertexPath;
        this.edgePath = edgePath;
        this.vertexType = vertexType;
        this.valueType = valueType;
        this.edgeValueType = edgeValueType;
        this.env = env;
    }
}
