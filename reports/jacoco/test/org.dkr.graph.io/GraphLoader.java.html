package org.dkr.graph.io;

import org.apache.flink.api.java.ExecutionEnvironment;
import org.apache.flink.graph.Graph;
import org.dkr.graph.algo.utils.GraphType;

/**
 * Graph generator for any types graph
 * @param <K> the vertex id data type
 * @param <VV> the vertex value data type
 * @param <EV> the edge value data type
 */
public class GraphLoader<K, VV, EV> {

    private final String vertexPath;
    private final String edgePath;
    private final ExecutionEnvironment env;
    private final Class<K> vertexType;
    private final Class<VV> valueType;
    private final Class<EV> edgeValueType;

    public GraphLoader(String vertexPath, String edgePath, ExecutionEnvironment env,
                       Class<K> vertexType, Class<VV> edgeType, Class<EV> edgeValueType) {
        this.vertexPath = vertexPath;
        this.edgePath = edgePath;
        this.vertexType = vertexType;
        this.valueType = edgeType;
        this.edgeValueType = edgeValueType;
        this.env = env;
    }

    /**
     * Generate a graph based on the graph type
     * @param type Graph type
     * @return graph object
     */
    public Graph<K, VV, EV> load(GraphType type) {

        Graph<K, VV, EV> graph = null;

        if (type == GraphType.DIRECT) {
            graph = new DirectGraph<>(this.vertexPath, this.edgePath, this.env,
                    this.vertexType, this.valueType, this.edgeValueType).generate();
        } else if (type == GraphType.UNDIRECT) {
            graph = new UndirectedGraph<>(this.vertexPath, this.edgePath, this.env,
                    this.vertexType, this.valueType, this.edgeValueType).generate();
        } else if (type == GraphType.WEIGHTED) {
            graph = new WeightedGraph<>(this.vertexPath, this.edgePath, this.env,
                    this.vertexType, this.valueType, this.edgeValueType).generate();
        }

        return graph;
    }
}
