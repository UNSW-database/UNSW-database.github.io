package org.dkr.graph.io;

import org.apache.flink.api.common.typeinfo.TypeInformation;
import org.apache.flink.api.common.typeinfo.Types;
import org.apache.flink.api.java.DataSet;
import org.apache.flink.api.java.ExecutionEnvironment;
import org.apache.flink.api.java.tuple.Tuple2;
import org.apache.flink.api.java.tuple.Tuple3;
import org.apache.flink.graph.Edge;
import org.apache.flink.graph.Graph;
import org.apache.flink.graph.Vertex;
import org.apache.flink.graph.generator.GraphGenerator;
import org.dkr.graph.algo.utils.GraphType;
import org.dkr.graph.io.utils.EdgeSplitter;
import org.dkr.graph.io.utils.VertexSplitter;

/**
 * Undirected graph generator
 * @param <K> the vertex id data type
 * @param <VV> the vertex value data type
 * @param <EV> the edge value data type
 */
public class UndirectedGraph<K, VV, EV> extends GraphBase<K, VV, EV> implements GraphGenerator<K, VV, EV> {

    public UndirectedGraph(String vertexPath, String edgePath, ExecutionEnvironment env,
                           Class<K> vertexType, Class<VV> valueType, Class<EV> edgeValueType) {
        super(vertexPath, edgePath, env, vertexType, valueType, edgeValueType);
    }

    /**
     * Read graph data, and generate the graph
     * @return Graph object
     */
    @Override
    public Graph<K, VV, EV> generate() {

        DataSet<Tuple2<K, VV>> vertices = env.readTextFile(this.vertexPath)
                .flatMap(new VertexSplitter<K, VV>(this.vertexType, this.valueType))
                .returns(Types.TUPLE(TypeInformation.of(vertexType), TypeInformation.of(valueType)));

        DataSet<Tuple3<K, K, EV>> edges = env.readTextFile(this.edgePath)
                .flatMap(new EdgeSplitter<K, EV>(this.vertexType, this.edgeValueType, GraphType.UNDIRECT))
                .returns(Types.TUPLE(TypeInformation.of(vertexType), TypeInformation.of(vertexType), TypeInformation.of(edgeValueType)));

        return Graph.fromTupleDataSet(vertices, edges, this.env);

    }


    @Override
    public GraphGenerator<K, VV, EV> setParallelism(int parallelism) {
        return null;
    }
}
